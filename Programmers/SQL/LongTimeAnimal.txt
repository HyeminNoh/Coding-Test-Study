오랜 기간 보호한 동물 문제(1)

SELECT NAME, DATETIME FROM ANIMAL_INS WHERE ANIMAL_ID NOT IN (SELECT ANIMAL_ID FROM ANIMAl_OUTS ) ORDER BY DATETIME LIMIT 3

오랜 기간 보호한 동물 문제(2)

SELECT OUTS.ANIMAL_ID, OUTS.NAME FROM ANIMAL_INS AS INS JOIN ANIMAL_OUTS AS OUTS ON INS.ANIMAL_ID = OUTS.ANIMAL_ID ORDER BY INS.DATETIME - OUTS.DATETIME LIMIT 2
